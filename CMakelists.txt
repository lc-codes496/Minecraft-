cmake_minimum_required(VERSION 3.16)
project(MinecraftCPP)

set(CMAKE_CXX_STANDARD 17)

# FetchContent para GLFW e GLAD (ou link manual)
include(FetchContent)
FetchContent_Declare(
  glfw
  GIT_REPOSITORY https://github.com/glfw/glfw.git
  GIT_TAG        3.3.8
)
FetchContent_MakeAvailable(glfw)

FetchContent_Declare(
  glad
  GIT_REPOSITORY https://github.com/Dav1dde/glad.git
  GIT_TAG        master
)
FetchContent_MakeAvailable(glad)

add_executable(MinecraftCPP src/main.cpp)

# include paths (assume glad provides include dir)
target_include_directories(MinecraftCPP PRIVATE ${glad_SOURCE_DIR}/include)
target_link_libraries(MinecraftCPP PRIVATE glfw glad)
if (WIN32)
  target_link_libraries(MinecraftCPP PRIVATE opengl32)
else()
  find_package(OpenGL REQUIRED)
  target_link_libraries(MinecraftCPP PRIVATE OpenGL::GL)
endif()
